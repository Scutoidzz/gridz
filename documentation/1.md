
### THE 6-HOUR SPRINT MANIFESTO

**THE MISSION:**
1. Create the three architecture files (`idt.hpp`, `pic.hpp`, `interrupts.asm`).
2. Fight the `Makefile` to compile the `.asm` file. 
3. Run the OS with the new files included, but keep the activation lines commented out so the current polling loop still works safely.

### FILE 1: `src/idt.hpp`
(Creates the 16-byte packed structs and the bit-shifting logic)

```cpp
#pragma once
#include <stdint.h>

struct idt_entry {
    uint16_t isr_low;
    uint16_t kernel_cs;
    uint8_t  ist;
    uint8_t  attributes;
    uint16_t isr_mid;
    uint32_t isr_high;
    uint32_t reserved;
} __attribute__((packed));

struct idtr {
    uint16_t limit;
    uint64_t base;
} __attribute__((packed));

__attribute__((aligned(0x10))) 
static struct idt_entry idt[256];

static struct idtr idtr_reg;

static inline void idt_set_descriptor(uint8_t vector, void* isr, uint8_t flags) {
    uint64_t descriptor = (uint64_t)isr;

    idt[vector].isr_low    = (uint16_t)(descriptor & 0xFFFF);
    idt[vector].kernel_cs  = 0x28; 
    idt[vector].ist        = 0;
    idt[vector].attributes = flags;
    idt[vector].isr_mid    = (uint16_t)((descriptor >> 16) & 0xFFFF);
    idt[vector].isr_high   = (uint32_t)((descriptor >> 32) & 0xFFFFFFFF);
    idt[vector].reserved   = 0;
}
```

### FILE 2: `src/pic.hpp`
(Sends the magic numbers to move the hardware interrupts)

```cpp
#pragma once
#include "io.hpp"

static inline void pic_init() {
    outb(0x20, 0x11);
    outb(0xA0, 0x11);
    outb(0x21, 0x20);
    outb(0xA1, 0x28);
    outb(0x21, 4);
    outb(0xA1, 2);
    outb(0x21, 0x01);
    outb(0xA1, 0x01);
    outb(0x21, 0x00);
    outb(0xA1, 0x00);
}
```

### FILE 3: `src/interrupts.asm`
(Saves the math, runs C++, restores the math)

```nasm
global keyboard_isr
extern keyboard_handler

keyboard_isr:
    push rax
    push rcx
    push rdx
    push r8
    push r9
    push r10
    push r11

    call keyboard_handler

    pop r11
    pop r10
    pop r9
    pop r8
    pop rdx
    pop rcx
    pop rax
    iretq
```
